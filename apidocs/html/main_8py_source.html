89 def emit(self, record):

92 This happens to be in the 6th frame upward.

98def get_logger():
99 """Override FastAPI logger with custom loguru."""
100 # Hook all other loggers into ours
101 logger_name_list = [name for name in logging.root.manager.loggerDict]
102 for logger_name in logger_name_list:
103 logging.getLogger(logger_name).setLevel(10)
104 logging.getLogger(logger_name).handlers = []
105 if logger_name == "sqlalchemy":
106 # Don't hook sqlalchemy, very verbose
107 continue
108 if "." not in logger_name:
109 logging.getLogger(logger_name).addHandler(InterceptHandler())
110
111 log.remove()
112 log.add(
113 sys.stderr,
114 level=settings.LOG_LEVEL,
115 format=(
116 "{time:YYYY-MM-DD HH:mm:ss.SSS} | {level: <8} "
117 "| {name}:{function}:{line} | {message}"
118 ),
119 enqueue=True, # Run async / non-blocking
120 colorize=True,
121 backtrace=True, # More detailed tracebacks
122 catch=True, # Prevent app crashes
123 )
124
125 # Only log to file in production
126 if not settings.DEBUG:
127 log.add(
128 "/opt/logs/fmtm.json",
129 level=settings.LOG_LEVEL,
130 enqueue=True,
131 serialize=True, # JSON format
132 rotation="00:00", # New file at midnight
133 retention="10 days",
134 # format=log_json_format, # JSON format func
135 )
136
137 log.add(
138 "/opt/logs/create_project.json",
139 level=settings.LOG_LEVEL,
140 enqueue=True,
141 serialize=True,
142 rotation="00:00",
143 retention="10 days",
144 filter=lambda record: record["extra"].get("task") == "create_project",
145 )

151@api.exception_handler(RequestValidationError)
152async def validation_exception_handler(request: Request, exc: RequestValidationError):
153 """Exception handler for more descriptive logging."""
154 errors = []
155 for error in exc.errors():
156 # TODO Handle this properly
157 if error["msg"] in ["Invalid input", "field required"]:
158 status_code = 422 # Unprocessable Entity

160 status_code = 400 # Bad Request
161 errors.append(
162 {
163 "loc": error["loc"],
164 "msg": error["msg"],
165 "error": error["msg"] + str([x for x in error["loc"]]),
166 }
167 )
168 return JSONResponse(status_code=status_code, content={"errors": errors})
