98def update_task_status(db: Session, user_id: int, task_id: int, new_status: TaskStatus):
99 if not user_id:
100 raise HTTPException(status_code=400, detail="User id required.")
101
102 db_user = user_crud.get_user(db, user_id, db_obj=True)
103 if not db_user:
104 raise HTTPException(
105 status_code=400, detail=f"User with id {user_id} does not exist."
106 )
107
108 db_task = get_task(db, task_id, db_obj=True)
109
110 if db_task:
111 if db_task.task_status in [TaskStatus.LOCKED_FOR_MAPPING, TaskStatus.LOCKED_FOR_VALIDATION]:
112 if not (
113 user_id is not db_task.locked_by
114 ):
115 raise HTTPException(
116 status_code=401,
117 detail=f"User {user_id} with username {db_user.username} has not locked this task.",
118 )
119
120 if verify_valid_status_update(db_task.task_status, new_status):
121 # update history prior to updating task
122 update_history = create_task_history_for_status_change(
123 db_task, new_status, db_user
124 )
125 db.add(update_history)
126
127 db_task.task_status = new_status
128
129 if new_status in [
130 TaskStatus.LOCKED_FOR_MAPPING,
131 TaskStatus.LOCKED_FOR_VALIDATION,
132 ]:
133 db_task.locked_by = db_user.id
134 else:
135 db_task.locked_by = None
136
137 if new_status == TaskStatus.MAPPED:
138 db_task.mapped_by = db_user.id
139 if new_status == TaskStatus.VALIDATED:
140 db_task.validated_by = db_user.id
141 if new_status == TaskStatus.INVALIDATED:
142 db_task.mapped_by = None
143
144 db.commit()
145 db.refresh(db_task)
146
147 return convert_to_app_task(db_task)
148
149 else:
150 raise HTTPException(
151 status_code=400,
152 detail=f"Not a valid status update: {db_task.task_status.name} to {new_status.name}",
153 )

232def convert_to_app_task(db_task: db_models.DbTask):
233 if db_task:
234 log.debug("")
235 log.debug(
236 f"Project ID {db_task.project_id} | Task ID "
237 f"{db_task.id} | Converting DB Task to App Task"
238 )
239 log.debug("")
240
241 app_task: tasks_schemas.Task = db_task
242 app_task.task_status_str = tasks_schemas.TaskStatusOption[
243 app_task.task_status.name
244 ]
245
246 if db_task.outline:
247 properties = {
248 "fid": db_task.project_task_index,
249 "uid": db_task.id,
250 "name": db_task.project_task_name,
251 }
252 log.debug("Converting task outline to geojson")
253 app_task.outline_geojson = geometry_to_geojson(
254 db_task.outline, properties, db_task.id
255 )
256 app_task.outline_centroid = get_centroid(db_task.outline)
257
258 if db_task.lock_holder:
259 app_task.locked_by_uid = db_task.lock_holder.id
260 app_task.locked_by_username = db_task.lock_holder.username
261 log.debug("Task currently locked by user " f"{app_task.locked_by_username}")
262
263 if db_task.qr_code:
264 log.debug(
265 f"QR code found for task ID {db_task.id}. Converting to base64"
266 )
267 app_task.qr_code_base64 = base64.b64encode(db_task.qr_code.image)
268 else:
269 log.warning(f"No QR code found for task ID {db_task.id}")
270 app_task.qr_code_base64 = ""
271
272 if db_task.task_history:
273 app_task.task_history = convert_to_app_history(db_task.task_history)
274
275 return app_task
276 else:
277 return None

280def convert_to_app_tasks(db_tasks: List[db_models.DbTask]):
281 num_tasks = len(db_tasks)

320async def update_task_files(
321 db: Session,
322 project_id: int,
323 project_odk_id: int,
324 project_name: str,
325 task_id: int,
326 category: str,
327 task_boundary: str,
328):
329 # This file will store osm extracts
330 task_polygons = f"/tmp/{project_name}_{category}_{task_id}.geojson"
331
332 # Update data extracts in the odk central
333 pg = PostgresClient("https://raw-data-api0.hotosm.org/v1", "underpass")
334
335 category = "buildings"
336
337 # This file will store osm extracts
338 outfile = f"/tmp/test_project_{category}.geojson"
339
340 # Delete all tasks of the project if there are some
341 db.query(db_models.DbFeatures).filter(
342 db_models.DbFeatures.task_id == task_id
343 ).delete()
344
345 # OSM Extracts
346 outline_geojson = pg.getFeatures(
347 boundary=task_boundary,
348 filespec=outfile,
349 polygon=True,
350 xlsfile=f"{category}.xls",
351 category=category,
352 )
353
354 updated_outline_geojson = {"type": "FeatureCollection", "features": []}
355
356 # Collect feature mappings for bulk insert
357 for feature in outline_geojson["features"]:
358 # If the osm extracts contents do not have a title, provide an empty text for that.
359 feature["properties"]["title"] = ""
360
361 feature_shape = shape(feature["geometry"])
362
363 wkb_element = from_shape(feature_shape, srid=4326)
364 updated_outline_geojson["features"].append(feature)
365
366 db_feature = db_models.DbFeatures(
367 project_id=project_id,
368 geometry=wkb_element,
369 properties=feature["properties"],
370 )
371 db.add(db_feature)
372 db.commit()
373
374 # Update task_polygons file containing osm extracts with the new geojson contents containing title in the properties.
375 with open(task_polygons, "w") as jsonfile:
376 jsonfile.truncate(0) # clear the contents of the file
377 dump(updated_outline_geojson, jsonfile)
378
379 # Update the osm extracts in the form.
380 central_crud.upload_xform_media(project_odk_id, task_id, task_polygons, None)
381
382 return True
