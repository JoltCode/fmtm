40 @field_validator("EXTRA_CORS_ORIGINS", mode="before")
41 @classmethod
42 def assemble_cors_origins(
43 cls,
44 val: Union[str, list[str]],
45 info: FieldValidationInfo,
46 ) -> Union[list[str], str]:
47 """Build and validate CORS origins list.
48
49 By default, the provided frontend URLs are included in the origins list.
50 If this variable used, the provided urls are appended to the list.
51 """
52 default_origins = []
53
54 # Build default origins from env vars
55 url_scheme = info.data.get("URL_SCHEME")
56 main_url = info.data.get("FRONTEND_MAIN_URL")
57 map_url = info.data.get("FRONTEND_MAP_URL")
58 if url_scheme and main_url and map_url:
59 default_origins = [
60 f"{url_scheme}://{main_url}",
61 f"{url_scheme}://{map_url}",
62 ]
63
64 if val is None:
65 return default_origins
66
67 if isinstance(val, str):
68 default_origins += [i.strip() for i in val.split(",")]
69 return default_origins
70
71 elif isinstance(val, list):
72 default_origins += val
73 return default_origins
74
75 raise ValueError(f"Not a valid CORS origin list: {val}")

86 @field_validator("FMTM_DB_URL", mode="after")
87 @classmethod
88 def assemble_db_connection(cls, v: Optional[str], info: FieldValidationInfo) -> Any:
89 """Build Postgres connection from environment variables."""
90 if isinstance(v, str):
91 return v
92 pg_url = PostgresDsn.build(
93 scheme="postgresql",
94 username=info.data.get("FMTM_DB_USER"),
95 password=info.data.get("FMTM_DB_PASSWORD"),
96 host=info.data.get("FMTM_DB_HOST"),
97 path=info.data.get("FMTM_DB_NAME", ""),
98 )
99 # Convert Url type to string
100 return str(pg_url)
