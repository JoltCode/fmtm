63def convert_to_app_users(db_users: List[db_models.DbUser]):
64 if db_users and len(db_users) > 0:
65 app_users = []
66 for user in db_users:
67 if user:
68 app_users.append(convert_to_app_user(user))
69 app_users_without_nones = [i for i in app_users if i is not None]
70 return app_users_without_nones
71 else:
72 return []
